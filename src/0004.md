# 简介

接口可以用来约束对象、函数、以及类的结构和类型，这是代码协作的契约，我们必须遵守并不能改变

# 对象类型接口

```javascript
// 从后端获取数据，然后渲染到页面中，我们就需要定义接口

interface List {
  readonly id: number;  // 只读属性
  name: string;
  // [x: string]: any;  // 字符串索引签名 -- 用任意的字符串去索引List，可以得到任意的结果，这样List就可以支持多个属性了 ，因为对象的key是string
  age?: number; // 可选属性，可以有可以没有
}

interface Result {
  data: List[]
}
// 渲染函数
function render(result: Result) {
  result.data.forEach((value) => {
    console.log(value.id, value.name)
    if (value.age) {
        console.log(value.age)
    }
    // value.id++   // 只读属性是不允许修改的
  })
}
// 这个就是我们从后端请求到数据，完全符合我们接口的定义
let result = {
  data: [
    {id: 1, name: 'A', sex: 'male'},  // 传入多余的sex，也不会报错，只要满足接口的必要条件就好
    {id: 2, name: 'B', age: 10}
  ]
}
// 渲染出来（也是这个函数的打印）
render(result)


// ===============
// 这样就会报错，需要改成上面那样，或者用下面的方法
render({data: [
  {id: 1, name: 'A', sex: 'male'},
  {id: 2, name: 'B', age: 10}
]})

// ============== 类型断言解决
render({
  data: [
  {id: 1, name: 'A', sex: 'male'},
  {id: 2, name: 'B', age: 10}
]} as Result)
// 这也是类型断言，不建议使用，在react中容易产生歧义
render(<Result>{
  data: [
  {id: 1, name: 'A', sex: 'male'},
  {id: 2, name: 'B', age: 10}
]})
```

用数字索引的结构

``` javascript
interface StringArray {
  [index: number]: string   // 用任意的数字去索引StringArray，都会得到一个string，这就相当于声明了一个字符串数组
}

let chars: StringArray = ['a', 'b']
```

用字符串去索引一个接口

``` javascript
interface Names {
  [x: string]: any;
  // y: number; // 上面定义了string所以，就不能定义number类型了
  [z: number]: number;  // 数字索引返回的类型，一定得是string索引返回的子级
}

```



# 函数类型接口
 
``` javascript
// 下面两种定义函数接口是等价的
let add: (x: number, y: number) => number
interface Add {
    (x: number, y: number): number
}

type Add = (x: number, y: number) => number // 类型别名定义函数
let add: Add = (a: number, b: number) => a + b  // 具体的实现
```

混合类型接口：即可以定义一个函数，也可以像对象一样拥有属性和方法

下面用混合类型定义一个类库

```javascript
interface Lib {
  (): void;   // 没有返回值也没有参数
  version: string;
  doSomething(): void;
}

// 实现一下上面的混合类型接口

function getLib() {
    let lib = (() => {}) as Lib
    lib.version = '1.0.0'
    lib.doSomething = () => {}
    return lib;
}
let lib1 = getLib()
lib1()
let lib2 = getLib()
lib2.doSomething()
```


# 如何定义一个函数

定义函数一共有四种方式

``` javascript
// function定义，参数类型需要指定，返回值可以利用类型推断省去
function add1(x: number, y: number) {
    return x + y
}
// 通过一个变量定义函数类型
let add2: (x: number, y: number) => number
// 通过类型别名定义函数类型
type add3 = (x: number, y: number) => number
// 通过接口定义函数类型
interface add4 {
    (x: number, y: number): number
}

// 后面三种只是函数类型的定义，并没有具体的实现，真正调用的时候需要书写这个函数体
```

``` javascript
// 在ts中形参和实参必须一一对应，下面这个就会报错
add1(1, 2, 3)

// y是可选参数就可以不传了,可选参数必须位于必须参数之后
function add5(x: number, y?: number) {
    return y ? x + y : x
}
add5(1)
```

函数的默认参数

``` javascript
function add6(x: number, y = 0, z: number, q = 1) {
    return x + y + z + q
}
// 必选参数前的默认参数必须明确传入undefined
add6(1, undefined, 3)
```

剩余参数: 剩余参数是以数组形式存在的
 
``` javascript
function add7(x: number, ...rest: number[]) {
    return x + rest.reduce((pre, cur) => pre + cur);
}
add7(1, 2, 3, 4, 5)
```

函数重载：函数名称相同，但是函数参数的类型或者个数不同

好处：不需要为了相似的函数，起不同的名字，增强了函数的可读性

``` javascript
// 重载列表
function add8(...rest: number[]): number;
function add8(...rest: string[]): string;
function add8(...rest: any[]) {
    let first = rest[0];
    if (typeof first === 'number') {
        return rest.reduce((pre, cur) => pre + cur);
    }
    if (typeof first === 'string') {
        return rest.join('');
    }
}
console.log(add8(1, 2))
console.log(add8('a', 'b', 'c'))
```
